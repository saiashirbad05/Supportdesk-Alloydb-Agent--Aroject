export BASE="https://supportdesk-nl-api-260584325793.us-central1.run.app"

# Colors
GRN='\033[0;32m'; YLW='\033[1;33m'; CYN='\033[0;36m'
MGN='\033[0;35m'; RED='\033[0;31m'; BLD='\033[1m'; NC='\033[0m'

clear
echo -e "${CYN}${BLD}"
echo "  ╔══════════════════════════════════════════════════════════════════╗"
echo "  ║       AlloyDB AI  —  Support Ticket Query System                ║"
echo "  ║       Live Deployment Status Report                             ║"
echo "  ╚══════════════════════════════════════════════════════════════════╝"
echo -e "${NC}"
echo -e "${YLW}  🌐  Service: ${BASE}${NC}"
echo ""

# ── Health Check ──────────────────────────────────────────────────────────────
echo -e "${MGN}  ┌─ ✅  HEALTH CHECK ─────────────────────────────────────────────┐${NC}"
HEALTH=$(curl -s "${BASE}/healthz")
DB=$(echo $HEALTH | python3 -c "import sys,json; d=json.load(sys.stdin); print(d.get('database','N/A'))" 2>/dev/null)
TS=$(echo $HEALTH | python3 -c "import sys,json; d=json.load(sys.stdin); print(d.get('timestamp','N/A')[:19])" 2>/dev/null)
ST=$(echo $HEALTH | python3 -c "import sys,json; d=json.load(sys.stdin); print(d.get('status','error'))" 2>/dev/null)
if [ "$ST" = "healthy" ]; then
  echo -e "${MGN}  │${NC}  Status    : ${GRN}● HEALTHY${NC}"
else
  echo -e "${MGN}  │${NC}  Status    : ${RED}● UNHEALTHY${NC}"
fi
echo -e "${MGN}  │${NC}  Database  : ${GRN}${DB}${NC}"
echo -e "${MGN}  │${NC}  Timestamp : ${DB:+${TS}}${NC}"
echo -e "${MGN}  └────────────────────────────────────────────────────────────────┘${NC}"
echo ""

# ── Query 1 ───────────────────────────────────────────────────────────────────
echo -e "${CYN}  ┌─ 🎫  QUERY 1 : Open High/Critical Payments Tickets (Mar 2026+) ┐${NC}"
Q1=$(curl -s "${BASE}/demo/payments-open")
CNT=$(echo $Q1 | python3 -c "import sys,json; d=json.load(sys.stdin); print(d.get('count',0))" 2>/dev/null)
echo -e "${CYN}  │${NC}  Total Matching Tickets : ${GRN}${BLD}${CNT}${NC}"
echo $Q1 | python3 -c "
import sys, json
d = json.load(sys.stdin)
for r in d.get('results', []):
    print(f\"  │  ▸ Ticket #{r['ticket_id']} | {r['customer_name']:<15} | {r['priority']:<8} | {r['status']:<12} | {r['assigned_engineer']}\")
" 2>/dev/null
echo -e "${CYN}  └────────────────────────────────────────────────────────────────┘${NC}"
echo ""

# ── Query 2 ───────────────────────────────────────────────────────────────────
echo -e "${YLW}  ┌─ 📊  QUERY 2 : Unresolved Tickets by Service ─────────────────┐${NC}"
Q2=$(curl -s "${BASE}/demo/unresolved-by-service")
echo $Q2 | python3 -c "
import sys, json
d = json.load(sys.stdin)
for r in d.get('results', []):
    bar = '█' * int(r['unresolved_count']) * 5
    print(f\"  │  ▸ {r['service_name']:<12} : {r['unresolved_count']} unresolved  {bar}\")
" 2>/dev/null
echo -e "${YLW}  └────────────────────────────────────────────────────────────────┘${NC}"
echo ""

# ── Query 3 ───────────────────────────────────────────────────────────────────
echo -e "${RED}  ┌─ 👨‍💻  QUERY 3 : Critical Open Auth Tickets ──────────────────────┐${NC}"
Q3=$(curl -s "${BASE}/demo/critical-auth")
echo $Q3 | python3 -c "
import sys, json
d = json.load(sys.stdin)
for r in d.get('results', []):
    print(f\"  │  ▸ Engineer : {r['assigned_engineer']}\")
    print(f\"  │    Ticket  : #{r['ticket_id']}\")
    print(f\"  │    Issue   : {r['issue_summary']}\")
" 2>/dev/null
echo -e "${RED}  └────────────────────────────────────────────────────────────────┘${NC}"
echo ""
