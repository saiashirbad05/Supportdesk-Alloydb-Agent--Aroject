import os
from flask import Flask, jsonify, request
import psycopg2
import psycopg2.extras

app = Flask(__name__)

DB_HOST = os.environ.get("INSTANCE_HOST", "127.0.0.1")
DB_PORT = int(os.environ.get("DB_PORT", "5432"))
DB_NAME = os.environ.get("DB_NAME", "supportdesk")
DB_USER = os.environ.get("DB_USER", "postgres")
DB_PASS = os.environ.get("DB_PASS", "")

def get_conn():
    return psycopg2.connect(
        host=DB_HOST,
        port=DB_PORT,
        dbname=DB_NAME,
        user=DB_USER,
        password=DB_PASS,
        connect_timeout=10
    )

@app.route("/", methods=["GET"])
def root():
    return jsonify({
        "service": "supportdesk-nl-api",
        "status": "running",
        "endpoints": [
            "/healthz",
            "/demo/payments-open",
            "/demo/unresolved-by-service",
            "/demo/critical-auth"
        ]
    }), 200

@app.route("/healthz", methods=["GET"])
def healthz():
    try:
        conn = get_conn()
        cur = conn.cursor()
        cur.execute("SELECT current_database(), now()")
        row = cur.fetchone()
        cur.close()
        conn.close()
        return jsonify({
            "status": "healthy",
            "database": row[0],
            "timestamp": str(row[1]),
            "db_host": DB_HOST
        }), 200
    except Exception as e:
        return jsonify({
            "status": "unhealthy",
            "error": str(e)
        }), 500

@app.route("/demo/payments-open", methods=["GET"])
def demo_payments_open():
    sql = """
        SELECT ticket_id, customer_name, service_name, priority, status,
               created_at::text AS created_at, assigned_engineer
        FROM app.support_tickets
        WHERE service_name = 'payments'
          AND status = 'Open'
          AND priority IN ('High', 'Critical')
          AND created_at >= TIMESTAMP '2026-03-01 00:00:00'
        ORDER BY created_at ASC
    """
    try:
        conn = get_conn()
        cur = conn.cursor(cursor_factory=psycopg2.extras.RealDictCursor)
        cur.execute(sql)
        rows = [dict(r) for r in cur.fetchall()]
        cur.close()
        conn.close()
        return jsonify({
            "query": "Show open high or critical tickets for payments service",
            "count": len(rows),
            "results": rows
        }), 200
    except Exception as e:
        return jsonify({"error": str(e)}), 500

@app.route("/demo/unresolved-by-service", methods=["GET"])
def demo_unresolved():
    sql = """
        SELECT service_name, COUNT(*) AS unresolved_count
        FROM app.support_tickets
        WHERE status IN ('Open', 'In Progress')
        GROUP BY service_name
        ORDER BY unresolved_count DESC
    """
    try:
        conn = get_conn()
        cur = conn.cursor(cursor_factory=psycopg2.extras.RealDictCursor)
        cur.execute(sql)
        rows = [dict(r) for r in cur.fetchall()]
        cur.close()
        conn.close()
        return jsonify({
            "query": "Unresolved tickets by service",
            "count": len(rows),
            "results": rows
        }), 200
    except Exception as e:
        return jsonify({"error": str(e)}), 500

@app.route("/demo/critical-auth", methods=["GET"])
def demo_critical_auth():
    sql = """
        SELECT assigned_engineer, ticket_id, issue_summary
        FROM app.support_tickets
        WHERE service_name = 'auth'
          AND priority = 'Critical'
          AND status = 'Open'
    """
    try:
        conn = get_conn()
        cur = conn.cursor(cursor_factory=psycopg2.extras.RealDictCursor)
        cur.execute(sql)
        rows = [dict(r) for r in cur.fetchall()]
        cur.close()
        conn.close()
        return jsonify({
            "query": "Critical auth tickets",
            "count": len(rows),
            "results": rows
        }), 200

        return jsonify({"error": str(e)}), 500

@app.errorhandler(404)
def not_found(e):
    return jsonify({
        "error": "Not found",
        "available_routes": [

            "/healthz",
            "/demo/payments-open",
            "/demo/unresolved-by-service",
            "/demo/critical-auth"
        ]

if __name__ == "__main__":
    port = int(os.environ.get("PORT", "8080"))
    app.run(host="0.0.0.0", port=port, debug=False)

SECTION 9: ALL ERRORS AND SOLUTIONS
================================================================================

ERROR 1: password complexity flag is required when public IP is enabled
------------------------------------------------------------------------
CAUSE: Creating instance with public IP without password complexity flag
FIX: Add --database-flags=password.enforce_complexity=on at instance creation


ERROR 2: server does not support SSL, but SSL was required
-----------------------------------------------------------
CAUSE: Used sslmode=require with alloydb-auth-proxy on localhost
FIX: Remove sslmode=require; proxy handles SSL automatically


ERROR 3: connection refused
----------------------------
CAUSE: Proxy not started or crashed
FIX: pkill -f alloydb-auth-proxy; restart proxy; wait 15 seconds


ERROR 4: column d.daticulocale does not exist
----------------------------------------------
CAUSE: Used psql -lqt on PostgreSQL 17
FIX: Query pg_database directly instead


ERROR 5: DB flag with name "alloydb.ai_nl_enabled" does not exist
------------------------------------------------------------------
CAUSE: Wrong flag name (dot instead of underscore)
FIX: Use alloydb_ai_nl.enabled=on


ERROR 6: ABORTED: unable to queue the operation
------------------------------------------------
CAUSE: Another operation already running on instance
FIX: Wait for instance state to be READY, then retry


ERROR 7: must be owner of extension google_ml_integration
----------------------------------------------------------
CAUSE: Ran ALTER EXTENSION google_ml_integration UPDATE as non-owner
FIX: Never run that command; skip it entirely


ERROR 8: schema "alloydb_ai_nl" does not exist
-----------------------------------------------
CAUSE: Extension created before flag took effect
FIX: Enable flag → restart proxy → then create extension


ERROR 9: Resource not found for the prediction request
-------------------------------------------------------
